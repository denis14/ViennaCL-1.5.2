00065         inline vcl_size_t next_power_2(vcl_size_t n) {
00066             n = n - 1;
00067 
00068             vcl_size_t power = 1;
00069 
00070             while(power < sizeof(vcl_size_t) * 8) {
00071                 n = n | (n >> power);
00072                 power *= 2;
00073             }
00074 
00075             return n + 1;
00076         }
